前台js

//滑动导航
    $('#nav .about li').shu_biao_yi_ru_yi_chu(function () {
        var target = $(this).sh_jd().offsetLeft;
        $('#nav .nav_bg').yi_dong_tou_ming({
            attr : 'x',
            target : target + 20,
            t : 30,
            step : 10,
            fn : function () {
                $('#nav .white').yi_dong_tou_ming({
                    attr : 'x',
                    target : -target
                });
            }
        });
    }, function () {
        $('#nav .nav_bg').yi_dong_tou_ming({
            attr : 'x',
            target : 20,
            t : 30,
            step : 10,
            fn : function () {
                $('#nav .white').yi_dong_tou_ming({
                    attr : 'x',
                    target : 0
                });
            }
        });
    });
