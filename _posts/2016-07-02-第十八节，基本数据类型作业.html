 2 # -*- coding:utf-8 -*-

 3 print("查找列表中元素，移除每个元素的空格，并查找以 a或A开头 并且以 c 结尾的所有元素。")

 4 li = ["aleb", " aric", "Alex", "Tony", "rain"]

10     if (b.startswith("a") or b.startswith("A")) and b.endswith("c"):

11         print(b) #打印出判断到的元素

12 #打印出

13 #aric
