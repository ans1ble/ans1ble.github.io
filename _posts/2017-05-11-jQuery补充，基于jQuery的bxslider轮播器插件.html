$(function () {
    //<!--普通轮播图-->
    $('.bxslider').bxSlider({auto: true, autoControls: true});

    //<!--横向旋转木马轮播-->
    $('.slider1').bxSlider({auto: true, slideWidth: 200, minSlides: 2, maxSlides: 3, slideMargin: 10});

    //<!--纵向旋转木马轮播-->
    $('.slider8').bxSlider({
        mode: 'vertical',
        slideWidth: 300,
        minSlides: 2,
        slideMargin: 10
    });
});
